\newglossaryentry{ai}{
    name={Intelligence Artificielle (IA)},
    description={Domaine de l'informatique visant à créer des machines capables de simuler l'intelligence humaine. L'IA englobe l'apprentissage automatique, la vision par ordinateur et le traitement du langage naturel.}
}

\newglossaryentry{etl}{
    name={ETL (Extract, Transform, Load)},
    description={Processus en informatique décisionnelle qui consiste à extraire des données de diverses sources, à les transformer pour répondre aux besoins de l'entreprise (nettoyage, agrégation, etc.) et à les charger dans un entrepôt de données ou une autre base de données cible.}
}

\newglossaryentry{bigdata}{
    name={Big Data},
    description={Ensemble de données volumineux qui dépasse les capacités des outils de gestion de bases de données traditionnels. Le Big Data est caractérisé par les "3V" : Volume, Vélocité et Variété.}
}
